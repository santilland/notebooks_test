Europe = ESDC_low_time_opt.sel(lat = slice(70.,30.), lon = slice(-20.,30.))
Europe_zscore = ((Europe-Europe.mean(dim='time'))/Europe.std(dim='time'))

ESRIN_zscore = Europe_zscore.sel(lon = 12.67,lat = 41.83, method = 'nearest')
